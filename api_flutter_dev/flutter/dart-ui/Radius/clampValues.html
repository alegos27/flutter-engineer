Radius clampValues({
  double? minimumX,
  double? minimumY,
  double? maximumX,
  double? maximumY,
}) {
  return Radius.elliptical(
    clampDouble(x, minimumX ?? -double.infinity, maximumX ?? double.infinity),
    clampDouble(y, minimumY ?? -double.infinity, maximumY ?? double.infinity),
  );
}
